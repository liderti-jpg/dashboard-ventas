-- Query actual (ítems en v400_item) SELECT i.f120_referencia AS referencia, RTRIM(i.f120_descripcion) AS descripcion, i.f120_id_tipo_inv_serv AS tipo_inventario, v.v400_cant_existencia_actual AS existencia_actual, v.v400_cant_disponible_1 AS disponible, v.v400_cant_comprometida_1 AS comprometido, v.v400_cant_pendiente_entrar_1 AS pendiente_entrar, v.v400_costo_prom_uni AS costo_prom_uni, v.v400_cant_existencia_actual * v.v400_costo_prom_uni AS valor_inventario, v.v400_nivel_minimo AS stock_minimo, v.v400_nivel_maximo AS stock_maximo, v.v400_punto_pedido AS punto_pedido, v.v400_abc_rotacion_veces AS abc_rotacion, v.v400_fecha_ult_venta AS fecha_ult_venta, v.v400_fecha_ult_entrada AS fecha_ult_entrada, v.v400_fecha_ult_salida AS fecha_ult_salida, DATEDIFF(DAY, v.v400_fecha_ult_salida, GETDATE()) AS dias_sin_salida, DATEDIFF(DAY, v.v400_fecha_ult_entrada, GETDATE()) AS dias_sin_entrada FROM v400_item v WITH (NOLOCK) INNER JOIN t120_mc_items i WITH (NOLOCK) ON v.v400_rowid_item = i.f120_rowid WHERE v.v400_rowid_bodega = 1 AND i.f120_id_cia = 1 UNION ALL -- Ítems BLOQUEADOS que no aparecen en v400_item SELECT i.f120_referencia AS referencia, RTRIM(i.f120_descripcion) AS descripcion, i.f120_id_tipo_inv_serv AS tipo_inventario, 0 AS existencia_actual, 0 AS disponible, 0 AS comprometido, 0 AS pendiente_entrar, i.f120_costo_prom AS costo_prom_uni, 0 AS valor_inventario, 0 AS stock_minimo, 0 AS stock_maximo, 0 AS punto_pedido, 'C' AS abc_rotacion, NULL AS fecha_ult_venta, NULL AS fecha_ult_entrada, NULL AS fecha_ult_salida, NULL AS dias_sin_salida, NULL AS dias_sin_entrada FROM t120_mc_items i WITH (NOLOCK) WHERE i.f120_id_cia = 1 AND i.f120_ind_estado = 'B' -- B = Bloqueado AND i.f120_rowid NOT IN ( SELECT v.v400_rowid_item FROM v400_item v WITH (NOLOCK) WHERE v.v400_rowid_bodega = 1 ) ORDER BY valor_inventario DESC FOR JSON PATH
